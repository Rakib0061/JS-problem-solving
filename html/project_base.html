# Input item sort and display


// @@@@@@@@@@@@@@aother exprement@@@@@@@@@@@@@@

let hyper = document.getElementById("hyper");

// ~~~~~~~~~~~~~~~~~~ creat element ~~~~~~~~~~~~~~~~~~

let ul = document.createElement("ul");
let input = document.createElement("input");
let btn = document.createElement("button");
btn.textContent = "Click";

// ~~~~~~~~~~~~~~ creat myPush prototype ~~~~~~~~~~~~~~

Array.prototype.myPush = function(...a){
this.push(a);
init();
}

// ~~~~~~~~~~~~~~~~ make li & li item ~~~~~~~~~~~~~~~~

let flower = ["rose","lily","sunflower","jasmine"];

const init = ()=>{
ul.innerHTML = "";
flower.sort().map(vlu =>{
let dad = vlu.toString().charAt(0).toUpperCase() + vlu.toString().slice(1).toLowerCase();

let li = document.createElement("li");
li.textContent = dad;
ul.appendChild(li);
});
};

init();

// ~~~~~~~~~~~~~~~~~~~ Click Event ~~~~~~~~~~~~~~~~~~~

btn.addEventListener("click",()=>{
flower.myPush(input.value);
input.value = "";
});

// ~~~~~~~~~~~~~ append parent tag ~~~~~~~~~~~~~

hyper.appendChild(ul);
hyper.appendChild(input);
hyper.appendChild(btn);
